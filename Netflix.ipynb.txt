# convert my duration to str
df['duration'] = df['duration'].astype(str)

# extract the minutes with regex
df['duration_min'] = df['duration'].str.extract(r'(\d+)\s*min', expand=False)
df['duration_min'] = pd.to_numeric(df['duration_min'], errors='coerce')

# extract the season in seasons with regex
df['duration_seasons'] = df['duration'].str.extract(r'(\d+)\s*Season', expand=False)
df['duration_seasons'] = pd.to_numeric(df['duration_seasons'], errors='coerce')
df.head()

# Show average duration for movies and average duration for seasons
df.groupby('type')[['duration_min', 'duration_seasons']].mean()

# Group movie by duration min
df.groupby('type')['duration_min'].mean().filter(like='Movie').round(2)

# Group TV Show by duration seasons
df.groupby('type')['duration_seasons'].mean().filter(like='TV Show').round(2)
